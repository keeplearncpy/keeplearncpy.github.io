# Dart学习

# 一、 Dart简介

· Dart是谷歌开发的，类型安全的，面向对象的编程语言，被应用于Web、服务器、移动应用和物联网等领域。

· Dart诞生于2011年10月10日

· 类似于Ts，是强类型语言

· 运行方式

  • 原生虚拟机（Dart代码可以运行在Windows、,Mac、linux上）
  • JS引擎（Dart代码可以转成Js代码，然后运行在浏览器上）
  • [https://pub.dev]Dart的包管理网站
  • pub对应于js的npm

# 二、如何绑定环境变量

  1. 先确定电脑上Flutter SDK的安装路径

    比如：C:\flutter

    那么C:\flutter\bin 就是flutter和dart的相关执行命令

    C:\flutter\bin\cache\dart-sdk\bin 就是Dart SDK的相关命令的执行路径

    C:\flutter\.pub-cache\bin 就是pub下载的公共模块的执行路径

# 四、Dart 语法基础

# 4.1. 简介

  • 注释语法与js一致
  • 声明函数不需要关键字（JS中通过function关键字来声明函数）
  • 函数和参数前面都有类型声明，void表示没有返回值，int是整型数字
  • 打印使用print（JS使用console.log()）
  • 每行代码结束时，必须以分号结束
  • 字符串通过引号包起来，支持模板字符串
  • main是入口函数，Dart应用程序总是从amin函数开始执行
  • 用var声明的变量，其数据类型是动态的

# 4.2. 语法基础 - 注释

  • 单行注释

    // 我是单行注释

  • 多行注释

    /我是多行注释/

  • 文档注释

    /// 我是文档注释 （可以通用dartdoc将注释转换成文档，文档注释支持md语法）

# 4.3. 语法基础 - 变量

  • 变量是一个引用，Dart万物皆对象，变量存储的时对象的引用
  • 声明变量
    • 明确指定类型： int age = 18
    • 不明确类型： var age = 18 或 dynamic age = 18
    • 类型推导 =》var/final/const/dynamic 在声明变量时就会有属于自己的类型
  • 变量名称大小写敏感（age与Age时两个不同的变量）
  • 变量默认值是null（JS中变量默认值是undefined）
  • Dart变量的值不会进行隐式转换（null不会自动转换成false）

# 4.44. 语法基础 - 常量

  • 常量就是值不可变的变量（一旦声明，其值不能更改）

  • 声明常量

    • const age = 18
    • final age = 18
    • const和final的区别 (const只能编译时，而final可以在运行时)
    • const time = DateTime.now() 报错--无法将运行是的值分配给const变量
    • final time = DateTime.now() 成功-可以将运行时的值分配给final变量 即可以通过计算/函数来获取一个值
void main() {
  // 声明变量
  var age = '18';
  print(age);

  // 指定数据类型
  String name = '张三';
  print(name);

  // 不知道类型的
  dynamic myname = 20;
  print(myname);

  // 变量的默认值
  var defaultName;
  print(defaultName);  //null
    
  // 常量
  const n1 = 10;
  print(n1); //常量一旦声明，就不能修改

  // final
  final n2 = 20;
  print(n2);
}

# 4.5. Dart数据类型 - Number

  • Dart中的数字由三个关键字描述

    • num表示数字类型（可整可小）
      • int（表示整数）
      • double（表示浮点数，可以整数，可以小数）

  • 常用API

    [https://api/dart.cn/stable/dart-core/num-class.html]

    [https://api/dart.cn/stable/dart-core/int-class.html]

    [https://api/dart.cn/stable/dart-core/double-class.html]

void main() {
  // 声明整数
  int count = 3;
  print(count);

  // 浮点数
  // double price = 7.15;
  double price = 7;
  print(price);

  // 声明数值类型
  num a1 = 3.7;
  print(a1); // num可以是小数可以是整数

  // 类型转换
  print(a1.toString()); //转换成字符串
  print(4.5.toInt()); //向下取整

  // 四舍五入
  print(3.1415.round());
  print(3.1415.toStringAsFixed(2)); //四舍五入几位

  // 返回余数
  print(10.remainder(3));

  // 数字比较:(前后相比)  0表示相等， 1 表示大于 -1 表示小于
  print(10.compareTo(50));
  // 反应会最大公约数
  print(12.gcd(18));
    
  // 科学计数法
  print(154300000.toStringAsExponential(2));
}

# 4.6. Dart数据类型-String

  • 声明字符串(String)
    • 单引号、双引号都可以
    • 三个单双引号可以让字符串换行
  • 常见API
    • [https://api.dart.cn/stable/dart-core/String-class.html]
  • 正则表达式
    • RegExp(r'正则表达式')
    • RegExp(r'\d+')
void main() {
  String str1 = "今天天气晴朗"; //单双都可以
  print(str1);

  // 三引号可以换行
  String str2 = '''今天天气真好，
  适合骑车''';
  print(str2);

  // 字符串拼接
  print(str1 + str2);
  print("$str1 $str2");

  // 字符串的分隔
  print(str1.split(''));

  // 字符串裁切
  print('  abc  '.trim());
  // 判断字符串是否为空
  print(''.isEmpty);
  print(''.isNotEmpty);

  // 字符串替换
  print(str1.replaceAll("天气晴朗", "倾盆大雨"));
  // 支持正则替换
  print('a1b2c3d5'.replaceAll(RegExp(r'\d+'), '_'));
  // 通过正则匹配手机号
  var isPhone = RegExp(r'^1\d{10}$');
  print(isPhone.hasMatch('12345678910'));
  print(isPhone.hasMatch('23456789101'));

  // 查找字符串
  print(str1.contains('气'));
  print(str1.indexOf('晴'));
}

# 4.8. Dart 数据类型 - List

  • Dart中的数组，由List对象表示，list有两种声明方式

    • 字面量方式
      • List list = []; //不限定元素的数据类型
      • List list = <int>[]; //限定元素的数据类型是int
    • 构造函数方式
      • List list = new List.empty(growable:true) //不嫌多长度的空列表
      • List list = new List.filled(3,0) //声明指定长度的填充列表

  • 扩展操作符(...)

    • var list = [1,2,3];
    • var list2 = [0,...list]; //[0,1,2,3]

  • 常用API

    [https://api.dart.cn/stable/dart-core/List-class.html]

void main() {
  List l1 = ['a', 'b', 'c', 1, 2, 3];
  print(l1);

  List l2 = <int>[1, 2, 3];
  print(l2);

  // 通过构造函数的声明方式
  var l3 = new List.empty(growable: true); //growable表示可以添加元素，是否具有成长性
  l3.add(1);
  print(l3);

  var l4 = new List.filled(3, 6);
  print(l4);

  // 扩展运算符
  var l5 = [0, ...l4];
  print(l5);

  var l6;
  var l7 = [0, ...?l6];
  print(l7);

  // 列表长度
  print(l1.length);

  // 列表的反转
  // reversed使用这个之后，要用toList()方法转换成列表
  print(l1.reversed.toList());

  // 添加元素
  l3.addAll([4, 5, 8]);
  print(l3);

  // 删除元素
  l3.remove(4);
  print(l3);

  // 删除指定位置的元素
  l3.removeAt(0);
  print(l3);

  // 清空列表
  l3.clear();
  print(l3.length);
  print(l3.isEmpty);

  // 合并列表
  List words = ['hello', 'dart'];
  print(words.join('-'));
}

# 4.8.1 遍历list

  • forEach()
    • 遍历列表
  • map()
    • 遍历并处理元素，然后生成新的列表
  • where()
    • 返回满足条件的数据
  • any()
    • 只要有一项满足条件，即返回true
  • every()
    • 判断是否每一项都满足条件，都满足条件才返回true
void main() {
  List nums = [1, 2, 3, 4];
  //for循环遍历
  for (var i = 0; i < nums.length; i++) {
    print(nums[i]);
  }

  // for in 遍历
  for (var i in nums) {
    print(i);
  }

  // forEach遍历
  nums.forEach((element) {
    print(element);
  });

  var newNums = [1, 2, 4];
  for (var i = 0; i < nums.length; i++) {
    newNums.add(nums[i] * 2);
  }
  var newNums1 = nums.map((e) {
    return e * 2;
  });
  print(newNums1.toList());  //[2, 4, 6, 8]

  // where
  var newNums2 = nums.where((element) => element % 2 == 0);
  print(newNums2.toList()); //[2, 4]

  // any()方法检测是否有奇数
  print(nums.any((element) => false)); //false

  // every()方法检测是否全部为偶数
  print(nums.every((element) => element % 2 == 0));  //false

  // 扩展
  var pairs = [
    [1, 2],
    [3, 4]
  ];
  var flatten = pairs.expand((pair) => pair).toList();
  print(flatten);
}  //[1, 2, 3, 4]

# 4.9. Dart数据类型 - Set

  • Set是一个无序的，元素唯一的集合

  • Set有字面量和构造函数两种声明方式（字面量中用大括号）

  • 无法通过下表取值

  • 具有集合特有的操作

    • 比如：求交集、并集、差集等

  • 常用API

    [https://api.dart.cn/stable/dart-core/Set-class.html]

void main() {
  var nums = <int>[1, 2, 3, 4];
  print(nums);

  // 通过构造函数的声明方式
  var fruits = new Set();
  fruits.add('苹果');
  fruits.add('香蕉');
  fruits.add('橘子');
  print(fruits);
  print(fruits.toList());

  List nums1 = [1, 2, 2, 4, 4, 5];
  print(nums1.toSet().toList()); // 去重

  // 集合特有操作
  var caocao = new Set();
  caocao.addAll(['张辽', '张郃', '关羽', '夏侯惇']);

  var liubei = new Set();
  liubei.addAll(['关羽', '张飞', '赵云', '马超']);

  // 交集
  print(caocao.intersection(liubei));

  // 并集
  print(caocao.union(liubei));

  // 差集
  print(caocao.difference(liubei));
}

# 4.10. Dart 数据类型 - Map

  • Map是一个无序的键值对(key-value)映射。通常被称作哈希或者字典

  • 声明方式

    • var map = new Map();
    • var map ={key1:value,key2:value};
    • map['key'] = value

  • 常用API

    [https://api.dart.cn/stable/dart-core/Map-class.html]

void main() {
  var person = {
    'name': '张三',
    'age': 20,
    'work': ['程序员', '送外卖']
  };
  print(person);

  // 通过构造函数的声明方式
  var person1 = new Map();
  person1['name'] = '李四';
  person1['age'] = 20;
  person1['work'] = ['程序员', '送外卖'];
  print(person1);

  // 访问属性
  print(person['name']);
}

# 五、Dart 函数的定义

# 5.1 函数的定义

Dart是一种真正的面向对象语言，所以即使函数也是对象，所有也有类型, 类型就是Function。

这也就意味着函数可以作为变量定义或者作为其他函数的参数或者返回值.

  • 函数的定义方式:
返回值类型 函数名(参数1, 参数2, ...) {
  函数体...
  return 返回值;
}

# 5.2. 函数的参数问题

函数的参数可以分成两类: 必须参数和可选参数.

  • 其定义方式为：
命名可选参数: {param1, param2, ...}
位置可选参数: [param1, param2, ...]

# 5.3. 函数是一等公民

在很多语言中, 函数并不能作为一等公民来使用, 比如Java/OC. 这种限制让编程不够灵活, 所以现代的编程语言基本都支持函数作为一等公民来使用, Dart也支持.

这就意味着你可以将函数赋值给一个变量, 也可以将函数作为另外一个函数的参数或者返回值来使用.

// 函数作为一等公民
void main() {
  var func = printHello;
  func(); // Hello
  print(func); // Closure: () => void from Function 'printHello': static.
  printHello1(printHello); // Hello
}

# 5.4. 匿名函数
