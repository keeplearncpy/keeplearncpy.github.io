# 一、 Java简介与安装

# 1. Java简介

  • Java是一种面向对象的编程语言，由Sun公司于1995年5月推出，是由James Gosling和同事们共同研发的，最初被命名为Oak，目标是为嵌入式消费电子产品提供一种革命性的程序设计语言。

# 三、 java基础

# 1. 基本类型

# 1.1 整数类型

  • byte：1个字节，8位，-128~127
  • short：2个字节，16位，-32768~32767
  • int：4个字节，32位，-2147483648~2147483647
  • long：8个字节，64位，-9223372036854775808~9223372036854775807

# 1.2 浮点类型 在java语言中，默认的浮点类型时double

  • float：4个字节，32位，-3.402823e+38~3.402823e+38 float有效位数是6-7位
  • double：8个字节，64位，-1.797693e+308~1.797693e+308 有效位数时15-16位

# 1.3 字符类型

  • char：2个字节，16位，0~65535

# 1.4 布尔类型

  • boolean：1个字节，8位，true/false

# 1.5 转义类型

  • \'：将单引号转为一个无意义的字符

  • \\：将反斜杠转为一个无意义的字符

  • \n：换行

  • \t：空格

  • 代码演示

package com.company;

public class Main {

    public static void main(String[] args) {
	// write your code here
    // java 中的基本数据类型
    // 整数类型
        int a = 30;
        long e = 40l;
        byte B = (byte)500;
        short S = 10;
        System.out.println(a);
        System.out.println(e);
        System.out.println(B);
        System.out.println(S);
    // 浮点类型   ----   在java语言中，默认的浮点类型时double
    //  在声明float类型的数据的时侯一定要加f或者F   ----- float有效位数时6-7位
        float c = 4.05629f;
        System.out.println(c);
    //  double类型数据 有效位数时15-16位
        double d = 5.63269;
        System.out.println(d);
    //  字符串  ----- char
        char ga = '你';
        System.out.println(ga);
        char gas = '你';
        System.out.println(gas);
    // 布尔类型
        boolean isShow = true;
        System.out.println(isShow);
        boolean isFale = false;
        System.out.println(isFale);
    // 转义类型 （重点）
        // 打印一个字符类型的数据 ’
        // \'将‘转为一个无意义的字符
        System.out.println('\'');
        // 打印一个\字符
        System.out.println('\\');

        // 几个特殊的符号
        //  \n 换行
        System.out.println("落霞与孤鹜齐飞\n秋水长天共一色");
        //  \t 空格
        System.out.println("仰天大笑出门去\t我辈岂是蓬篙人");


    // 声明一个字符串数据
    // String引用数据类型
        String str = "长风破浪会有时\n直挂云帆济沧海";
        System.out.println(str);
    }
}

  • ==：等于

# 3.4 逻辑运算符

  • &&：与 --- 一假全假
  • ||：或 --- 一真全真
  • !：非
  • 代码演示：
package com.company;

public class RelationalOperators {
    // 类体
    public static void main(String[] args) {
        boolean res = 1 > 2;
        System.out.println(res); //false
        boolean res1 = 5 != 4;
        System.out.println(res1); //true

        // 逻辑 或 与
        boolean res2 = 3 > 4 && 5 > 1;
        System.out.println(res2);   //false //一假全假
        boolean res3 = 5 > 6 || 10 >2;
        System.out.println(res3);  //true  //一真全真

        // 非
        boolean res4 = !(4 > 5);
        System.out.println(res4);

        // 逻辑运算符短路原则
        // 逻辑与的短路原则：
        // 条件1 && 条件2  && 条件3
        // 只要有一个为false，直接为false，后面的条件直接不执行
        // 逻辑或的短路原则
        // 条件1 || 条件2 || 条件3
        // 只要有一个为true，后面条件直接不执行

        int num1 = 5;
        // 逻辑或 在第一个为true时，后面的直接不执行
        // 一般开发中，逻辑或和逻辑与是单独使用的
        boolean res5 = (3 > 1) || (++num1 > 10) && (4 > 3);
        System.out.println(res5);
        System.out.println(num1);

    }
}
