# 一. 联合类型与交叉类型

# 1. 联合类型

TS 中的类型系统允许我们使用多种运算符，联合类型就是其中之一

  1. 含义：联合类型是由两个或者多个其他类型组成的类型，它表示标识符可以是这几种类型里面其中之一，联合类型中的每一个类型被称之为联合成员
  2. 举例：
function getLength(arg: number | string | array) {
  if ((typeof arg === "string") | (typeof arg === "array")) {
    console.log(arg.length);
  } else {
    console.log(info.toFixed(2));
  }
}

// 补充：
// 类型缩小  ---就是当目标标识符有多个类型时，可以使用类型缩小进行某些特定的操作
let info: string | number = "abc";
info = 123;
//使用时---可以使用类型缩小
if (typeof info === "string") {
  console.log(info.length);
} else {
  console.log(info.toFixed(2));
}

# 4.类型别名和接口的区别

  1. 如果是定义非对象类型，通常推荐使用 type，比如 Direction、Alignment、一些 Function
  2. interface 可以重复的对某个接口来定义属性和方法，type 不行（定义对象）
  3. interface 可以 extends(继承)，type 不行

# 5. 交叉类型

  1. 含义：交叉类型是由多个类型组成的类型，它表示标识符同时拥有多种类型的成员
  2. 语法
type NewName = Type1 & Type2 & Type3;
1
  1. 举例：
interface A {
  x: number;
  foo(bar: number): number; // 返回值类型为number
}
interface B {
  y: string;
  foo(bar: string): string; // 返回值类型为string
}
type C = A & B; // C的类型为A和B的交叉类型
let c: C = {
  x: 1,
  y: "2",
  foo(bar: any) {
    return bar;
  },
};

//或者
interface IName {
  name: string;
}
interface IAge {
  age: number;
  code: (x: number) => void;
}
type IPerson = IName & IAge;
const person: IPerson = {
  name: "tom",
  age: 18,
  code: function (x) {
    console.log(x);
  },
};
console.log(person);

# 6. 类型断言 as

  1. 语法：<类型>值 或者 值 as 类型
  2. 作用：将一个联合类型的变量，指定为一个更加具体的类型
  3. 注意：类型断言不是类型转换，断言成一个联合类型中不存在的类型是不允许的
  4. 举例：
//类型断言
interface IName {
  name: string;
}
interface IAge {
  age: number;
  code: (x: number) => void;
}
type IPerson = IName | IAge;
const person: IPerson = {
  name: "tom",
  age: 18,
  code: function (x) {
    console.log(x);
  },
};
console.log(person);
//类型断言
(person as IName).name = "jerry";
console.log(person);

# 7. 非空类型断言 !

  1. 语法：!
  2. 作用：用来断言一个变量是非空类型，即排除 null 和 undefined
  3. 注意：使用非空类型断言，一定要确定该变量是非空的，否则会报错

    非空断言使用的是 ! ，表示可以确定某个标识符是有值的，跳过 ts 在编译阶段对它的检测；

//非空类型断言
interface IName {
  age: number;
  name: string;
  friend: {
    name: string;
  };
}
const person: IName = {
  name: "tom",
  age: 18,
  friend: {
    name: "jerry",
  },
};
//非空类型断言
console.log(person.friend.name); //报错，可能不存在
person.friend?.name = "jack"; //报错，赋值时，等号左边不能是由可选属性
person.friend!.name = "jack"; //不报错，非空类型断言

# 8.字面量类型

  1. 含义：字面量类型就是一个值的集合，这个值只能是这个集合中的某一个值
  2. 语法：使用|连接多个字面量类型

//字面量类型
type Direction = "up" | "down" | "left" | "right";
function move(direction: Direction) {
  console.log(direction);

# 9. 类型缩小

  1. 含义：类型缩小就是在某些条件下，将一个联合类型的变量，缩小为一个更加具体的类型
  2. 语法：使用typeof、instanceof、in、字面量类型等运算符
  3. 举例：
//类型缩小---typeof
let info: string | number = "abc";
info = 123;
//使用时---可以使用类型缩小
if (typeof info === "string") {
  console.log(info.length);
} else {
  console.log(info.toFixed(2));
}

// 平等缩小 switch == ！== ===
enum Direction {
  UP,
  DOWN,
  LEFT,
  RIGHT,
}

const d1: Direction = Direction.DOWN;
function trunDirection(direction: Direction) {
  switch (direction) {
    case Direction.LEFT:
      console.log("向左移动一步");
      break;
    case Direction.RIGHT:
      console.log("向右移动一步");
      break;
  }
}

// instanceof操作符
function fn(val: string | number) {
  if (val instanceof String) {
    console.log(val.length);
  } else {
    console.log(val.toFixed(2));
  }
}

// in操作符
type IName = {
  name: string;
  age: number;
};
type IAge = {
  habby: string;
};
type IPerson = IName | IAge;
function fn2(person: IPerson) {
  if ("name" in person) {
    console.log(person.name);
  } else {
    console.log(person.age);
  }
}

# 二. TS 函数编程

# 1. 函数类型

  1. 含义：函数类型是由参数类型和返回值类型组成的类型，它表示标识符可以作为函数调用
  2. 语法：(参数:类型,参数:类型,...)=>返回值类型
  3. 举例：
//函数类型
type Fn = (x: number, y: number) => number;
const fn: Fn = function (x, y) {
  return x + y;
};
console.log(fn(1, 2));

# 2.调用签名

  1. 含义：调用签名是由参数类型和返回值类型组成的类型，它表示标识符可以作为函数调用

# 3.构造签名

  1. 含义：构造签名是由参数类型和返回值类型组成的类型，它表示标识符可以作为构造函数调用
  2. 语法：new (参数:类型,参数:类型,...)=>返回值类型
  3. 举例：
//在TS中使用class不仅创建了一个类，还创建了一个构造函数
class Person {}
interface ICreactPerson {
  //构造签名 new ():返回类型
  new (): Person;
}
function factory(fn: ICreactPerson) {
  const f = new fn();
  return f;
}
factory(Person);

# 4. 可选参数

  1. 含义：可选参数就是在参数名后面加上一个?，表示该参数可以传递也可以不传递
  2. 语法：参数名?:类型
  3. 举例：
//可选参数num2是一个联合类型 可以为number或者undefined
function sum(num1: number, num2?: number) {
  //   console.log(num1 + num2); //报错，num2可能为undefined
  // 使用类型缩小
  if (num2 !== undefined) {
    console.log(num1 + num2);
  } else {
    console.log(num1);
  }
}

sum(10);
sum(10, 50);
sum(10, undefined);

# 5. 默认参数

  1. 含义：默认参数就是在参数名后面加上一个=值，表示该参数可以传递也可以不传递，如果不传递就使用默认值
  2. 语法：参数名:类型=值

// 函数的参数可以有默认值

function foo(x: number, y = 100) {
  console.log(y + 10);
  console.log(x + y);
}

foo(50, undefined);

# 6. 剩余参数

  1. 含义：剩余参数就是在参数名前面加上一个...，表示该参数可以接受多个值，多个值会被放到一个数组中
  2. 语法：...参数名:类型
  3. 举例：
// 剩余参数
function sum(...args: number[]) {
  console.log(args);
  let total = 0;
  for (let i = 0; i < args.length; i++) {
    total += args[i];
  }
  return total;
}
console.log(sum(1, 2, 3, 4, 5));

# 7. 函数重载

function add(arg1: number, arg2: number): number;
function add(arg1: string, arg2: string): string;

// 3.2 编写通用的函数实现
function add(arg1: any, arg2: any): any {
  return arg1 + arg2;
}

add(10, 20);
add("aa", "bb");
// 通用函数不能被调用

# 9.this 的内置工具

  1. 含义：this 的内置工具就是用来描述函数中的 this 类型的
  2. 语法：使用 this 的内置工具，需要在 tsconfig.json 中配置 lib 选项
  3. 举例：
//this: { name: string } 这个不算第一个参数
function foo(this: { name: string }, info: { name: string }) {
  console.log(this, info);
}
type FooType = typeof foo;
//1. ThisParameterType:获取FooType函数中this的类型
type FooThisType = ThisParameterType<FooType>;
//2. OmitThisParameter:获取FooType函数中除了this之外的参数类型
type FooOmitThisType = OmitThisParameter<FooType>;
//3. ThisType:获取FooType函数中this的类型  ---- 用于指定函数中this的类型
interface IState {
  name: string;
  age: number;
}
interface IStore {
  state: IState;
  eating: () => void;
  running: () => void;
}
const store: IStore & ThisType<IState> = {
  state: {
    name: "why",
    age: 18,
  },
  eating() {
    console.log(this.name);
  },
  running() {
    console.log(this.age);
  },
};
export {};
