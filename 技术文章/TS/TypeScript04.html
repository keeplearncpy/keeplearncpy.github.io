# 一、 TS-映射类型

// 映射类型

# 1.应用场景

  1. 有时候，一个类型需要给予另外一个类型，但是不想要拷贝一份，或者拷贝一份会造成大量重复代码，这个时候可以考虑使用映射类型
  • 大部分内置的工具都是通过映射类型来实现的
  • 大多数类型体操的题目也是通过映射类型完成的
  1. 映射类型建立在索引签名的语法上：
  • 映射类型，就是使用了 PropertyKeys 联合类型的泛型
  • 其中 PropertyKeys 多事通过 keyof 创建，然后循环遍历键名创建一个类型
  • ts 提供了映射类型： 可以把它当成一个函数
  • 映射类型不能使用 interface 定义

# 2. 简介

  1. 映射类型是一种类型操作，它会将一个类型中的每个属性都转换为新的类型，然后组成一个新的类型

# 3. 语法

// 语法
type NewType = { [PropertyKeys in keyof OldType]: NewType };
1
2

# 4. 举例（映射类型-修饰符符号使用）

type MapPerson<T> = {
  readonly // 索引签名   -----相当于循环遍历传入过来的对象属性与对象值
  //   readonly 表示只读
  // ？表示可选
  [Property in keyof T]?: T[Property];
  // 可以通过前缀 - 或者 + 删除或者添加这些修饰符，如果没有写前缀，相当于使用了 + 前缀。
  // [Property in keyof T]-?: T[Property];
  //   相当于
  // name:string ....
};

interface Person {
  name: string;
  age: number;
}

type newPerson = MapPerson<Person>;
export {};

# 5. 举例（映射类型-修饰符符号使用）

//  当原来的类型里面的个别属性有readonly 和? 时，这个时候可以使用映射类型修饰符符号
//  - readonly -？ 表示去除里面的相关修饰符，默认是+号

  -readonly // 索引签名   -----相当于循环遍历传入过来的对象属性与对象值
  //   readonly 表示只读
  // ？表示可选
  [Property in keyof T]-?: T[Property];
  //   相当于
  // name:string ....
};

interface Person {
  name: string;
  age: number;
  readonly habby: string;
  address?: string;
}

type newPerson = MapPerson<Person>;
export {};

# 二、 TS-条件类型

日常开发中我们需要基于输入的值来决定输出的值，同样我们也需要基于输入的值的类型来决定输出的值的类型。 条件类型就是用来帮助我们描述输入类型和输出类型之间的关系。其写法类似于 js 中的条件表达式（三元表达式）

# 1. 语法

// 语法
type NewType = T extends U ? X : Y;
1
2

# 2. 基本使用

// 条件类型提供了 infer 关键词，可以从正在比较的类型中推断类型，然后再true分支里引用该推断结果

// 获取一个函数的参数类型和返回值类型
type IDType = number | string;
type ResType<T> = T extends IDType ? true : false;

// 举个例子
// function sum(a: number, b: number): number;
// function sum(a: string, b: string): string;

// 错误做法：类型扩大化
// function sum (a:number | string, b:number | string): number | string

function sum<T extends number | string>(
  a: T,
  b: T
): T extends number ? number : string;
function sum(a, b) {
  return a + b;
}

const res1 = sum(1, 2);
const res2 = sum("1", "2");
// const res3 = sum(1, '2'); // error

// infer 关键词
type ParamType<T> = T extends (param: infer P) => any ? P : T;
type Param = ParamType<typeof sum>; // Param = number | string

export {};

# 3. infer 关键词

  1. 条件类型提供了 infer 关键词，可以从正在比较的类型中推断类型，然后在 true 分支里引用该推断结果
  2. infer 关键词可以用在 extends 条件语句中，用来推断待推断的类型
type CalcFnType = (num1: number, num2: string) => number;

function foo() {
  return "abc";
}

// 总结类型体操题目: MyReturnType
type MyReturnType<T extends (...args: any[]) => any> = T extends (
  ...args: any[]
) => infer R
  ? R
  : never;

type MyParameterType<T extends (...args: any[]) => any> = T extends (
  ...args: infer A
) => any
  ? A
  : never;

// 获取一个函数的返回值类型: 内置工具
type CalcReturnType = MyReturnType<CalcFnType>;
type FooReturnType = MyReturnType<typeof foo>;
// type FooReturnType2 = MyReturnType<boolean>

type CalcParameterType = MyParameterType<CalcFnType>;

export {};

# 4. 条件类型的分发

  1. 条件类型的分发：条件类型中的类型变量如果是一个联合类型，那么这个条件类型就会被拆分成多个条件类型
  2. 举例：
// 条件类型的分发
// 条件类型中的类型变量如果是一个联合类型，那么这个条件类型就会被拆分成多个条件类型
// 举例：
type T1<T> = T extends string ? string : number;
type T2 = T1<string | number>;
// T2 = T1<string> | T1<number>
// T2 = string | number

// 或
type toArray<T> = T extends any ? T[] : never;

// number[]|string[]

type NumArray = toArray<number>;

// number[]|string[] 而不是 (number|string)[]
type NumAndStrArray = toArray<number | string>;

export {};

# 5. Pick<T,K>

// 4. Pick<T,K> 从传入的属性中挑选一个或者多个属性
interface Person {
  name: string;
  age: number;
  slogan?: string;
}

// 类型体操
// name | age | slogan
type keys = keyof Person;
type Res = keyof any; // => number|string|symbol

// 确定keys一定是可以作为key的联合类型
type HYRecord<Keys extends keyof any, T> = {
  [P in Keys]: T;
};

// Person都变成可选的
type t1 = "张三" | "李四" | "王五";
type Persons = HYRecord<t1, Person>;

const Persons: Persons = {
  张三: {
    name: "xxx",
    age: 10,
  },
  李四: {
    name: "yyy",
    age: 52,
  },
  王五: {
    name: "zzz",
    age: 34,
  },
};

export {};

# 6. Omit<T,K>

//  Omit<T,K> 从传入的属性中删除一个或者多个属性
interface Person {
  name: string;
  age: number;
  slogan?: string;
}
// Omit用来确定哪些属性可以忽略
// 确定keys一定是可以作为key的联合类型
type HYOmit<T, K extends keyof T> = {
  [P in keyof T as P extends K ? never : P]: T[P];
};

// Person都变成可选的
type Persons = HYOmit<Person, "slogan" | "name">;

export {};

# 10. InstanceType

//  InstanceType<T> 获取构造函数类型的实例类型
class Person {}
class Dog {}

// 类型体操
type HYInstanceType<T extends new (...args: any[]) => any> = T extends new (
  ...args: any[]
) => infer R
  ? R
  : never;

const p1: Person = new Person();

// typeof Person: 构造函数具体的类型
// InstanceType构造函数创建出来的实例对象的类型
type HYPerson = HYInstanceType<typeof Person>;
const p2: HYPerson = new Person();

// 构造函数的例子
// 通过的创建实例的工具函数时会用到这个InstanceType
function factory<T extends new (...args: any[]) => any>(
  ctor: T
): HYInstanceType<T> {
  return new ctor();
}

const p3 = factory(Person);
const d = factory(Dog);
export {};
